CREATE TABLE `employee_extra_work_tbl` ( `work_id` INT(11) NOT NULL AUTO_INCREMENT, `employee_id` INT(11) NOT NULL, -- Foreign key to staff_tbl.id `project_name` VARCHAR(255) NOT NULL, `task` VARCHAR(255) NOT NULL, `work_description` TEXT NOT NULL, `hours_worked` DECIMAL(5,2) NOT NULL, `work_date` DATE NOT NULL, `assigned_by` VARCHAR(255) DEFAULT NULL, -- Name or ID of the person who assigned the work `work_status` ENUM('not_yet_started', 'on_going', 'completed', 'pending') DEFAULT 'not_yet_started', -- Updated statuses `notes` TEXT DEFAULT NULL, -- Optional: Any additional notes about the extra work `added_on` DATETIME DEFAULT CURRENT_TIMESTAMP, `updated_on` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP, PRIMARY KEY (`work_id`), FOREIGN KEY (`employee_id`) REFERENCES `staff_tbl`(`id`) ON DELETE CASCADE ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
